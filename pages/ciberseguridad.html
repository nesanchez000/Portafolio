Aplicaciones prácticas en Python

Aplicación de análisis de logs (Mini SIEM)

import re
from collections import Counter

with open("logs.txt") as f:
    data = f.readlines()

ips = []
fails = 0

for line in data:
    if "FAILED LOGIN" in line:
        fails += 1
        ip = re.search(r"IP (\d+\.\d+\.\d+\.\d+)", line)
        if ip:
            ips.append(ip.group(1))

print("Intentos fallidos:", fails)
print("Top IPs sospechosas:", Counter(ips).most_common(3))

Qué aprendí

  • Identificar patrones en logs reales
  • Detectar intentos de fuerza bruta
  • Extraer IPs sospechosas con regex
  • Interpretar eventos de seguridad

Aplicación de auditoría OWASP

import requests

def analizar(url):
    r = requests.get(url)

    print("HTTPS:", url.startswith("https"))
    print("X-Frame-Options:", r.headers.get("X-Frame-Options"))
    print("Content-Security-Policy:", r.headers.get("Content-Security-Policy"))
    print("Cookies inseguras:", any("Secure" not in c for c in r.cookies))

analizar("https://example.com")

Qué aprendí

  • Detectar configuraciones inseguras en aplicaciones web
  • Analizar cabeceras HTTP críticas
  • Identificar cookies vulnerables
  • Aplicar principios OWASP

Aplicación de análisis de contraseñas

import hashlib

def analizar_pass(p):
    score = 0
    if len(p) >= 10: score += 1
    if any(c.isdigit() for c in p): score += 1
    if any(c.isupper() for c in p): score += 1
    if any(c in "!@#$%^&*" for c in p): score += 1

    print("Seguridad:", score, "/ 4")
    print("Hash SHA256:", hashlib.sha256(p.encode()).hexdigest())

analizar_pass("Nuria2024!")

Qué aprendí

  • Evaluar la fortaleza de contraseñas
  • Aplicar criterios de seguridad
  • Generar hashes seguros
  • Buenas prácticas de autenticación
