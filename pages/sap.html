Ejercicios prácticos en ABAP

SELECT + LOOP (Clientes España)

REPORT z_select_clientes.

DATA: lt_clientes TYPE TABLE OF kna1,
      ls_cliente  TYPE kna1.

SELECT *
  FROM kna1
  INTO TABLE lt_clientes
  WHERE land1 = 'ES'.

LOOP AT lt_clientes INTO ls_cliente.
  WRITE: / ls_cliente-kunnr,
           ls_cliente-name1,
           ls_cliente-ort01.
ENDLOOP.

UPDATE de material (ejercicio controlado)

REPORT z_update_material.

DATA lv_matnr TYPE matnr VALUE '000000000123456789'.

UPDATE mara
  SET mtart = 'ZMAT'
  WHERE matnr = lv_matnr.

IF sy-subrc = 0.
  WRITE: 'Material actualizado correctamente'.
ELSE.
  WRITE: 'No se pudo actualizar el material'.
ENDIF.

FORM + PERFORM (Generar texto)

REPORT z_form_ejemplo.

DATA lv_text TYPE string.

PERFORM generar_text USING 'Núria Elies' CHANGING lv_text.

WRITE: / lv_text.

FORM generar_text USING iv_nombre TYPE string
                   CHANGING ev_text TYPE string.
  CONCATENATE 'Hola,' iv_nombre 'bienvenida a SAP.'
         INTO ev_text SEPARATED BY space.
ENDFORM.

BAPI Material Save (simulación)

REPORT z_bapi_material.

DATA: ls_material TYPE bapi_mara,
      lt_return   TYPE TABLE OF bapiret2.

ls_material-matl_type = 'ZMAT'.
ls_material-matl_group = '001'.

CALL FUNCTION 'BAPI_MATERIAL_SAVEDATA'
  EXPORTING
    headdata = ls_material
  TABLES
    return   = lt_return.

LOOP AT lt_return INTO DATA(ls_msg).
  WRITE: / ls_msg-type, ls_msg-message.
ENDLOOP.

ALV Grid Display (Materiales)

REPORT z_alv_mara.

DATA: lt_mara TYPE TABLE OF mara.

SELECT *
  FROM mara
  INTO TABLE lt_mara
  UP TO 50 ROWS.

CALL FUNCTION 'REUSE_ALV_GRID_DISPLAY'
  EXPORTING
    i_structure_name = 'MARA'
  TABLES
    t_outtab         = lt_mara.
